CREATE TABLE Customer (
    CustomerID NUMBER PRIMARY KEY,
    Name VARCHAR2(50),
    Age NUMBER,
    Balance NUMBER,
    IsVIP VARCHAR2(5)
);
CREATE TABLE Loan (
    LoanID NUMBER PRIMARY KEY,
    CustomerID NUMBER,
    InterestRate NUMBER,
    DueDate DATE,
    FOREIGN KEY (CustomerID)
    REFERENCES Customer(CustomerID)
);

BEGIN
   FOR rec IN (
      SELECT l.LoanID
      FROM Customer c
      JOIN Loan l
      ON c.CustomerID = l.CustomerID
      WHERE c.Age > 60
   )
   LOOP
      UPDATE Loan
      SET InterestRate = InterestRate - 1
      WHERE LoanID = rec.LoanID;
   END LOOP;

END;

BEGIN

      SELECT CustomerID
      FROM Customer
      WHERE Balance > 10000
   )
   LOOP
      UPDATE Customer
      SET IsVIP = 'TRUE'
      WHERE CustomerID = rec.CustomerID;
   END LOOP;

END;

      WHERE l.DueDate BETWEEN SYSDATE

   END LOOP;
END;
